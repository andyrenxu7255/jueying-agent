### 1.1 Workflow 创建 - 生产级实现

```typescript
// services/workflow/src/repositories/workflow.repository.ts
import { Pool, PoolClient, QueryResult } from 'pg';
import { Logger } from 'winston';
import {
  WorkflowAlreadyExistsError,
  WorkflowCreationError,
  DatabaseConnectionError
} from '../errors/workflow.errors';
import { CreateWorkflowDTO, Workflow } from '../types/workflow.types';

export class WorkflowRepository {
  constructor(
    private readonly pool: Pool,
    private readonly logger: Logger
  ) {}

  async create(
    dto: CreateWorkflowDTO,
    client?: PoolClient
  ): Promise<Workflow> {
    const startTime = Date.now();
    const connection = client || await this.pool.connect();

    try {
      // 前置校验：检查必填字段
      this.validateCreateDTO(dto);

      // 检查是否已存在（幂等性保证）
      const existing = await this.findById(dto.id, connection);
      if (existing) {
        this.logger.warn('Workflow already exists, returning existing', {
          workflowId: dto.id,
          actor: dto.actor_id
        });
        return existing;
      }

      // 开始事务（如果未提供外部连接）
      if (!client) {
        await connection.query('BEGIN');
      }

      // 插入主记录
      const result = await connection.query<QueryResult<Workflow>>(
        `INSERT INTO workflow_instance (
          id, user_id, channel_type, workflow_type,
          status, priority, timeout_sec, policy_snapshot_hash,
          created_at, updated_at
        ) VALUES ($1, $2, $3, $4, $5, $6, $7, $8, NOW(), NOW())
        RETURNING *`,
        [
          dto.id,
          dto.user_id,
          dto.channel_type,
          dto.workflow_type || 'knowledge-query',
          'draft',
          dto.priority || 'normal',
          dto.timeout_sec || 3600,
          dto.policy_snapshot_hash
        ]
      );

      if (!result.rows || result.rows.length === 0) {
        throw new WorkflowCreationError(
          'Failed to insert workflow record',
          { workflowId: dto.id }
        );
      }

      const workflow = result.rows[0];

      // 初始化 Stage 记录（基于模板）
      await this.initializeStages(workflow.id, dto.stage_plan, connection);

      // 提交事务
      if (!client) {
        await connection.query('COMMIT');
      }

      this.logger.info('Workflow created successfully', {
        workflowId: workflow.id,
        durationMs: Date.now() - startTime,
        stageCount: dto.stage_plan?.length || 0
      });

      return workflow;

    } catch (error) {
      // 回滚事务
      if (!client) {
        try {
          await connection.query('ROLLBACK');
        } catch (rollbackError) {
          this.logger.error('Failed to rollback transaction', {
            error: rollbackError.message,
            originalError: error.message
          });
        }
      }

      // 错误分类与包装
      if (error.code === '23505') { // PostgreSQL unique_violation
        throw new WorkflowAlreadyExistsError(
          `Workflow with id ${dto.id} already exists`,
          { workflowId: dto.id, pgError: error }
        );
      }

      if (error.code === 'ECONNREFUSED' || error.code === '57P01') {
        throw new DatabaseConnectionError(
          'Database connection failed',
          { originalError: error }
        );
      }

      // 记录详细错误信息
      this.logger.error('Failed to create workflow', {
        workflowId: dto.id,
        error: error.message,
        stack: error.stack,
        durationMs: Date.now() - startTime
      });

      throw new WorkflowCreationError(
        'Unexpected error during workflow creation',
        { workflowId: dto.id, cause: error },
        error
      );

    } finally {
      // 确保释放连接（仅当是内部创建的连接）
      if (!client) {
        connection.release();
      }
    }
  }

  private validateCreateDTO(dto: CreateWorkflowDTO): void {
    const errors: string[] = [];

    if (!dto.id || typeof dto.id !== 'string' || dto.id.length === 0) {
      errors.push('workflow id is required and must be non-empty string');
    }

    if (!dto.user_id || typeof dto.user_id !== 'string') {
      errors.push('user_id is required');
    }

    if (!dto.policy_snapshot_hash || typeof dto.policy_snapshot_hash !== 'string') {
      errors.push('policy_snapshot_hash is required for security');
    }

    // 验证优先级枚举值
    const validPriorities = ['low', 'normal', 'high', 'critical'];
    if (dto.priority && !validPriorities.includes(dto.priority)) {
      errors.push(`priority must be one of: ${validPriorities.join(', ')}`);
    }

    // 验证超时值范围
    if (dto.timeout_sec && (dto.timeout_sec < 60 || dto.timeout_sec > 86400)) {
      errors.push('timeout_sec must be between 60 and 86400 seconds');
    }

    if (errors.length > 0) {
      throw new WorkflowCreationError(
        'Validation failed for workflow creation',
        { validationErrors: errors, input: dto }
      );
    }
  }

  private async initializeStages(
    workflowId: string,
    stagePlan: Array<{ stage_key: string; seq: number }> | undefined,
    client: PoolClient
  ): Promise<void> {
    if (!stagePlan || stagePlan.length === 0) {
      this.logger.warn('No stage plan provided, using default template', {
        workflowId
      });
      // 使用默认模板...
      return;
    }

    try {
      for (const stage of stagePlan) {
        await client.query(
          `INSERT INTO workflow_stage (
            id, workflow_id, stage_key, seq, status,
            created_at, updated_at
          ) VALUES (
            gen_random_uuid(), $1, $2, $3, 'pending', NOW(), NOW()
          )`,
          [workflowId, stage.stage_key, stage.seq]
        );
      }
    } catch (error) {
      this.logger.error('Failed to initialize stages', {
        workflowId,
        stageCount: stagePlan.length,
        error: error.message
      });
      throw new WorkflowCreationError(
        'Failed to initialize workflow stages',
        { workflowId, cause: error },
        error
      );
    }
  }

  async findById(id: string, client?: PoolClient): Promise<Workflow | null> {
    try {
      const result = await (client || this.pool).query<QueryResult<Workflow>>(
        'SELECT * FROM workflow_instance WHERE id = $1 AND deleted_at IS NULL',
        [id]
      );
      return result.rows[0] || null;
    } catch (error) {
      this.logger.error('Failed to find workflow by id', {
        workflowId: id,
        error: error.message
      });
      return null;
    }
  }
}
```

### 1.2 自定义错误类体系

```typescript
// common/errors/base-error.ts

export enum ErrorCategory {
  VALIDATION = 'VALIDATION',
  AUTHORIZATION = 'AUTHORIZATION',
  NOT_FOUND = 'NOT_FOUND',
  CONFLICT = 'CONFLICT',
  INTEGRATION = 'INTEGRATION',
  INTERNAL = 'INTERNAL'
}

export interface ErrorContext {
  [key: string]: unknown;
}

export class AppError extends Error {
  public readonly statusCode: number;
  public readonly errorCode: string;
  public readonly category: ErrorCategory;
  public readonly context: ErrorContext;
  public readonly timestamp: string;
  public readonly retryable: boolean;

  constructor(
    message: string,
    options: {
      statusCode?: number;
      errorCode?: string;
      category?: ErrorCategory;
      context?: ErrorContext;
      cause?: Error;
      retryable?: boolean;
    } = {}
  ) {
    super(message);

    this.name = this.constructor.name;
    this.statusCode = options.statusCode ?? 500;
    this.errorCode = options.errorCode ?? 'SYSTEM_INTERNAL_ERROR';
    this.category = options.category ?? ErrorCategory.INTERNAL;
    this.context = options.context ?? {};
    this.timestamp = new Date().toISOString();
    this.retryable = options.retryable ?? false;

    if (options.cause) {
      this.cause = options.cause;
    }

    // 确保原型链正确（用于 instanceof 检查）
    Object.setPrototypeOf(this, new.target.prototype);
  }

  toJSON(): Record<string, unknown> {
    return {
      name: this.name,
      message: this.message,
      statusCode: this.statusCode,
      errorCode: this.errorCode,
      category: this.category,
      context: this.context,
      timestamp: this.timestamp,
      retryable: this.retryable,
      ...(this.cause ? { cause: this.cause.message } : {})
    };
  }
}

// 具体错误类示例
export class WorkflowCreationError extends AppError {
  constructor(message: string, context?: ErrorContext, cause?: Error) {
    super(message, {
      statusCode: 422,
      errorCode: 'WORKFLOW_PLANNING_FAILED',
      category: ErrorCategory.INTERNAL,
      context,
      cause,
      retryable: false
    });
  }
}

export class WorkflowNotFoundError extends AppError {
  constructor(workflowId: string) {
    super(`Workflow not found: ${workflowId}`, {
      statusCode: 404,
      errorCode: 'WORKFLOW_NOT_FOUND',
      category: ErrorCategory.NOT_FOUND,
      context: { workflowId },
      retryable: false
    });
  }
}
```

## 2. LLM Provider 调用 - 生产级实现

```typescript
// services/llm/providers/openai.adapter.ts
import OpenAI from 'openai';
import { Logger } from 'winston';
import {
  LLMRequest,
  LLMResponse,
  LLMAdapter,
  LLMRateLimitError,
  LLMTimeoutError,
  LLMAuthenticationError
} from './types';

interface RetryConfig {
  maxRetries: number;
  baseDelayMs: number;
  maxDelayMs: number;
  retryableErrors: string[];
}

export class OpenAIAdapter implements LLMAdapter {
  private readonly client: OpenAI;
  private readonly logger: Logger;
  private readonly defaultRetryConfig: RetryConfig;

  constructor(
    config: {
      apiKey: string;
      baseURL?: string;
      organization?: string;
      timeout?: number;
    },
    logger: Logger
  ) {
    this.validateConfig(config);

    this.client = new OpenAI({
      apiKey: config.apiKey,
      baseURL: config.baseURL || 'https://api.openai.com/v1',
      organization: config.organization,
      timeout: config.timeout || 120_000,
      maxRetries: 0, // 我们自己控制重试逻辑
    });

    this.logger = logger.child({ provider: 'openai' });

    this.defaultRetryConfig = {
      maxRetries: 3,
      baseDelayMs: 1000,
      maxDelayMs: 30_000,
      retryableErrors: [
        'rate_limit_exceeded',
        'server_error',
        'timeout'
      ]
    };
  }

  private validateConfig(config: { apiKey?: string }): void {
    const errors: string[] = [];

    if (!config.apiKey || config.apiKey.trim().length < 10) {
      errors.push('API key is required and must be at least 10 characters');
    }

    if (errors.length > 0) {
      throw new LLMAuthenticationError(
        'Invalid OpenAI configuration',
        { validationErrors: errors }
      );
    }
  }

  async complete(request: LLMRequest): Promise<LLMResponse> {
    const requestId = `req_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`;
    const startTime = Date.now();

    this.logger.info('LLM request started', {
      requestId,
      modelId: request.model_id,
      messageCount: request.messages?.length,
      hasTools: !!request.tools?.length
    });

    let lastError: Error | null = null;

    for (let attempt = 0; attempt <= this.defaultRetryConfig.maxRetries; attempt++) {
      try {
        if (attempt > 0) {
          const delay = this.calculateBackoff(attempt);
          this.logger.info(`Retrying LLM request (attempt ${attempt + 1})`, {
            requestId,
            delayMs: delay,
            lastError: lastError?.message
          });
          await this.sleep(delay);
        }

        const response = await this.executeWithTimeout(request, requestId);

        this.logger.info('LLM request completed successfully', {
          requestId,
          attempt: attempt + 1,
          modelId: response.model_id,
          promptTokens: response.usage?.input_tokens,
          completionTokens: response.usage?.output_tokens,
          durationMs: Date.now() - startTime,
          finishReason: response.finish_reason
        });

        return response;

      } catch (error: unknown) {
        lastError = error as Error;

        const isRetryable = this.isRetryableError(error);

        this.logger.warn('LLM request attempt failed', {
          requestId,
          attempt: attempt + 1,
          error: error instanceof Error ? error.message : String(error),
          isRetryable,
          remainingAttempts: this.defaultRetryConfig.maxRetries - attempt
        });

        if (!isRetryable || attempt >= this.defaultRetryConfig.maxRetries) {
          break;
        }
      }
    }

    // 所有重试均失败，抛出最终错误
    this.logger.error('LLM request failed after all retries', {
      requestId,
      totalAttempts: this.defaultRetryConfig.maxRetries + 1,
      finalError: lastError?.message,
      durationMs: Date.now() - startTime
    });

    throw this.classifyFinalError(lastError!, requestId);
  }

  private async executeWithTimeout(
    request: LLMRequest,
    requestId: string
  ): Promise<LLMResponse> {
    try {
      const params: OpenAI.Chat.CompletionCreateParams = {
        model: request.model_id,
        messages: request.messages.map(msg => ({
          role: msg.role as 'user' | 'assistant' | 'system',
          content: msg.content
        })),
        temperature: request.temperature ?? 0.7,
        max_tokens: request.max_tokens,
        stop: request.stop_sequences,
        ...(request.tools?.length ? {
          tools: request.tools.map(tool => ({
            type: 'function' as const,
            function: {
              name: tool.name,
              description: tool.description,
              parameters: tool.parameters
            }
          })),
          tool_choice: request.tool_choice
        } : {})
      };

      const completion = await this.client.chat.completions.create(params);

      return {
        request_id: requestId,
        model_id: completion.model || request.model_id,
        content: completion.choices[0]?.message?.content || '',
        tool_calls: completion.choices[0]?.message?.tool_calls?.map(tc => ({
          id: tc.id,
          name: tc.function.name,
          arguments: tc.function.arguments
        })),
        usage: {
          input_tokens: completion.usage?.prompt_tokens || 0,
          output_tokens: completion.usage?.completion_tokens || 0
        },
        finish_reason: completion.choices[0]?.finish_reason || 'stop'
      };

    } catch (error: unknown) {
      const openaiError = error as { status?: number; code?: string; message?: string };

      if (openaiError.status === 401 || openaiError.status === 403) {
        throw new LLMAuthenticationError(
          'OpenAI authentication failed',
          { status: openaiError.status, message: openaiError.message }
        );
      }

      if (openaiError.status === 429) {
        throw new LLMRateLimitError(
          'OpenAI rate limit exceeded',
          {
            status: openaiError.status,
            retryAfter: this.parseRetryAfter(openaiError)
          }
        );
      }

      if (openaiError.code === 'timeout' || this.isTimeoutError(error)) {
        throw new LLMTimeoutError(
          'OpenAI request timeout',
          { modelId: request.model_id }
        );
      }

      throw error;
    }
  }

  private calculateBackoff(attempt: number): number {
    const exponentialDelay =
      this.defaultRetryConfig.baseDelayMs * Math.pow(2, attempt - 1);
    const jitter = Math.random() * 1000; // 添加随机抖动避免惊群效应
    return Math.min(exponentialDelay + jitter, this.defaultRetryConfig.maxDelayMs);
  }

  private isRetryableError(error: unknown): boolean {
    if (error instanceof LLMRateLimitError ||
        error instanceof LLMTimeoutError) {
      return true;
    }

    const err = error as { code?: string; status?: number };
    return this.defaultRetryConfig.retryableErrors.includes(err.code || '') ||
           (err.status && err.status >= 500 && err.status < 600);
  }

  private classifyFinalError(error: Error, requestId: string): Error {
    if (error instanceof LLMRateLimitError ||
        error instanceof LLMTimeoutError ||
        error instanceof LLMAuthenticationError) {
      return error;
    }

    return new Error(
      `LLM request failed after all retries: ${error.message}`,
      { cause: error }
    );
  }

  private parseRetryAfter(error: { headers?: { get?: (name: string) => string | null } }): number {
    const retryAfter = error.headers?.get?.('retry-after');
    if (retryAfter) {
      const seconds = parseInt(retryAfter, 10);
      if (!isNaN(seconds)) {
        return seconds * 1000;
      }
    }
    return this.defaultRetryConfig.baseDelayMs;
  }

  private isTimeoutError(error: unknown): boolean {
    return error instanceof Error &&
      (error.message.includes('timeout') ||
       error.message.includes('ETIMEDOUT') ||
       error.message.includes('AbortError'));
  }

  private sleep(ms: number): Promise<void> {
    return new Promise(resolve => setTimeout(resolve, ms));
  }

  count_tokens(text: string): number {
    // 简单估算：大约每4个字符一个token（英文）或每1.5个字符一个token（中文）
    const chineseChars = (text.match(/[\u4e00-\u9fa5]/g) || []).length;
    const otherChars = text.length - chineseChars;
    return Math.ceil(chineseChars / 1.5 + otherChars / 4);
  }

  get_model_info(modelId: string): ModelInfo {
    const modelMap: Record<string, ModelInfo> = {
      'gpt-4o': {
        model_id: 'gpt-4o',
        model_name: 'gpt-4o-2024-08-06',
        context_window: 128_000,
        max_output_tokens: 16_384,
        capabilities: ['planning', 'code', 'reasoning']
      },
      'gpt-4o-mini': {
        model_id: 'gpt-4o-mini',
        model_name: 'gpt-4o-mini-2024-07-18',
        context_window: 128_000,
        max_output_tokens: 16_384,
        capabilities: ['chat', 'light-reasoning']
      }
    };

    return modelMap[modelId] || (() => {
      throw new Error(`Unknown model: ${modelId}`);
    })();
  }
}
```

## 3. 配置管理器 - 生产级实现

```typescript
// common/config/config-manager.ts
import * as fs from 'fs';
import * as path from 'path';
import { Logger } from 'winston';
import Ajv from 'ajv';
import { ConfigSchema, ConfigDiff } from './types';

interface Watcher {
  keys: string[];
  callback: (config: ConfigSchema, diff: ConfigDiff) => void;
}

export class ConfigManager {
  private config: ConfigSchema;
  private watchers: Watcher[] = [];
  private readonly logger: Logger;
  private readonly schemaValidator: Ajv;
  private configFilePath: string;
  private lastModifiedTime: number = 0;

  constructor(
    private readonly defaultConfig: ConfigSchema,
    logger: Logger
  ) {
    this.config = { ...defaultConfig };
    this.logger = logger.child({ component: 'ConfigManager' });
    this.schemaValidator = new Ajv({ allErrors: true, strict: false });
  }

  async loadConfig(configDir?: string): Promise<ConfigSchema> {
    this.logger.info('Loading configuration...');

    try {
      const layers = [
        { name: 'defaults', data: this.defaultConfig },
        { name: 'file', data: await this.loadFromFile(configDir) },
        { name: 'environment', data: this.loadFromEnvironment() }
      ];

      let mergedConfig: Record<string, unknown> = {};

      for (const layer of layers) {
        if (layer.data) {
          this.logger.debug(`Applying config layer: ${layer.name}`, {
            keys: Object.keys(layer.data)
          });
          mergedConfig = this.deepMerge(mergedConfig, layer.data);
        }
      }

      this.config = mergedConfig as ConfigSchema;

      const validationResult = this.validateConfig(this.config);
      if (!validationResult.valid) {
        throw new Error(
          `Configuration validation failed:\n${JSON.stringify(validationResult.errors, null, 2)}`
        );
      }

      this.logger.info('Configuration loaded successfully', {
        configKeys: Object.keys(this.config),
        nodeEnv: this.config.server?.node_env
      });

      return this.config;

    } catch (error) {
      this.logger.error('Failed to load configuration', {
        error: error instanceof Error ? error.message : String(error),
        stack: error instanceof Error ? error.stack : undefined
      });

      if (process.env.NODE_ENV === 'production') {
        throw new Error(
          'Cannot start without valid configuration in production mode',
          { cause: error }
        );
      }

      this.logger.warn('Falling back to default configuration (non-production)');
      this.config = { ...this.defaultConfig };
      return this.config;
    }
  }

  async reload(): Promise<void> {
    this.logger.info('Reloading configuration...');

    const oldConfig = { ...this.config };

    try {
      const newConfig = await this.loadConfig();

      const diff = this.diffConfig(oldConfig, newConfig);

      if (diff.changed.length === 0) {
        this.logger.debug('No configuration changes detected');
        return;
      }

      if (diff.requiresRestart) {
        this.logger.warn('Configuration change requires restart', {
          changedKeys: diff.changed,
          reason: diff.restartReason
        });

        this.emitConfigChangeEvent(newConfig, diff);
        return;
      }

      this.config = newConfig;
      this.notifyWatchers(diff);

      this.logger.info('Configuration reloaded successfully', {
        changedKeys: diff.changed,
        requiresRestart: false
      });

    } catch (error) {
      this.logger.error('Failed to reload configuration', {
        error: error instanceof Error ? error.message : String(error)
      });

      throw error;
    }
  }

  watch(keys: string[], callback: (config: ConfigSchema, diff: ConfigDiff) => void): void {
    this.watchers.push({ keys, callback });
    this.logger.debug(`Registered config watcher for keys: ${keys.join(', ')}`);
  }

  getConfig(): ConfigSchema {
    return { ...this.config };
  }

  get<T>(path: string, defaultValue?: T): T {
    const value = path.split('.').reduce((obj: unknown, key: string) => {
      if (obj && typeof obj === 'object' && key in obj) {
        return (obj as Record<string, unknown>)[key];
      }
      return undefined;
    }, this.config);

    return (value as T) ?? defaultValue!;
  }

  private async loadFromFile(configDir?: string): Promise<Record<string, unknown> | null> {
    const possiblePaths = [
      configDir && path.join(configDir, `${process.env.NODE_ENV || 'development'}.yaml`),
      configDir && path.join(configDir, 'default.yaml'),
      'config/default.yaml'
    ].filter(Boolean) as string[];

    for (const filePath of possiblePaths) {
      try {
        if (fs.existsSync(filePath)) {
          const stats = fs.statSync(filePath);
          this.lastModifiedTime = stats.mtime.getTime();

          const content = fs.readFileSync(filePath, 'utf-8');

          const yaml = await import('js-yaml');
          const parsed = yaml.load(content) as Record<string, unknown>;

          this.logger.info(`Loaded config from file: ${filePath}`);
          return parsed;
        }
      } catch (error) {
        this.logger.warn(`Failed to load config from ${filePath}`, {
          error: error instanceof Error ? error.message : String(error)
        });
      }
    }

    return null;
  }

  private loadFromEnvironment(): Record<string, unknown> {
    const envConfig: Record<string, unknown> = {};

    const envMappings: Record<string, string[]> = {
      'server.port': ['SERVER_PORT', 'PORT'],
      'server.host': ['SERVER_HOST'],
      'database.url': ['DATABASE_URL'],
      'database.poolSize': ['DATABASE_POOL_SIZE'],
      'redis.url': ['REDIS_URL'],
      'workflow.maxConcurrent': ['WORKFLOW_MAX_CONCURRENT'],
      'executor.maxConcurrentSessions': ['EXECUTOR_MAX_CONCURRENT_SESSIONS'],
      'logging.level': ['LOG_LEVEL']
    };

    for (const [configPath, envVars] of Object.entries(envMappings)) {
      for (const envVar of envVars) {
        const value = process.env[envVar];
        if (value !== undefined) {
          this.setNestedValue(envConfig, configPath, this.parseEnvValue(value));
          this.logger.debug(`Loaded env var ${envVar} -> ${configPath}`);
          break;
        }
      }
    }

    return envConfig;
  }

  private setNestedValue(obj: Record<string, unknown>, path: string, value: unknown): void {
    const keys = path.split('.');
    let current: Record<string, unknown> = obj;

    for (let i = 0; i < keys.length - 1; i++) {
      const key = keys[i];
      if (!(key in current) || typeof current[key] !== 'object' || current[key] === null) {
        current[key] = {};
      }
      current = current[key] as Record<string, unknown>;
    }

    current[keys[keys.length - 1]] = value;
  }

  private parseEnvValue(value: string): unknown {
    if (value === 'true') return true;
    if (value === 'false') return false;
    if (/^\d+$/.test(value)) return parseInt(value, 10);
    if (/^\d+\.\d+$/.test(value)) return parseFloat(value);
    return value;
  }

  private deepMerge(target: Record<string, unknown>, source: Record<string, unknown>): Record<string, unknown> {
    const result = { ...target };

    for (const key of Object.keys(source)) {
      if (
        source[key] &&
        typeof source[key] === 'object' &&
        !Array.isArray(source[key]) &&
        target[key] &&
        typeof target[key] === 'object' &&
        !Array.isArray(target[key])
      ) {
        result[key] = this.deepMerge(
          target[key] as Record<string, unknown>,
          source[key] as Record<string, unknown>
        );
      } else {
        result[key] = source[key];
      }
    }

    return result;
  }

  private validateConfig(config: ConfigSchema): { valid: boolean; errors: string[] | null } {
    try {
      // 这里应该加载 JSON Schema 并验证
      // 简化版本：只做基本检查
      const errors: string[] = [];

      if (config.server?.port && (config.server.port < 1 || config.server.port > 65535)) {
        errors.push(`Invalid server port: ${config.server.port}`);
      }

      if (config.database?.poolSize && config.database.poolSize < 1) {
        errors.push('Database pool size must be at least 1');
      }

      return {
        valid: errors.length === 0,
        errors: errors.length > 0 ? errors : null
      };

    } catch (error) {
      return {
        valid: false,
        errors: [`Validation error: ${error instanceof Error ? error.message : String(error)}`]
      };
    }
  }

  private diffConfig(oldConfig: ConfigSchema, newConfig: ConfigSchema): ConfigDiff {
    const changed: string[] = [];
    const restartRequiredKeys = ['database.url', 'redis.url'];

    const compareObjects = (
      prefix: string,
      oldObj: Record<string, unknown>,
      newObj: Record<string, unknown>
    ): void => {
      const allKeys = new Set([...Object.keys(oldObj), ...Object.keys(newObj)]);

      for (const key of allKeys) {
        const fullPath = prefix ? `${prefix}.${key}` : key;
        const oldValue = oldObj[key];
        const newValue = newObj[key];

        if (JSON.stringify(oldValue) !== JSON.stringify(newValue)) {
          changed.push(fullPath);
        }
      }
    };

    compareObjects('', oldConfig as unknown as Record<string, unknown>, newConfig as unknown as Record<string, unknown>);

    const requiresRestart = changed.some(key =>
      restartRequiredKeys.some(reqKey => key.startsWith(reqKey))
    );

    return {
      changed,
      requiresRestart,
      restartReason: requiresRestart
        ? 'Database or Redis configuration changed'
        : undefined
    };
  }

  private notifyWatchers(diff: ConfigDiff): void {
    for (const watcher of this.watchers) {
      if (watcher.keys.some(key => diff.changed.some(changedKey =>
        changedKey.startsWith(key) || key.startsWith(changedKey)
      ))) {
        try {
          watcher.callback(this.config, diff);
        } catch (error) {
          this.logger.error('Config watcher callback failed', {
            keys: watcher.keys,
            error: error instanceof Error ? error.message : String(error)
          });
        }
      }
    }
  }

  private emitConfigChangeEvent(config: ConfigSchema, diff: ConfigDiff): void {
    this.logger.emit('config.change.requiring_restart', {
      config,
      diff,
      timestamp: new Date().toISOString()
    });
  }
}
```

## 4. 中间件 - 错误处理与日志

```typescript
// common/middleware/error-handler.middleware.ts
import { Request, Response, NextFunction } from 'express';
import { Logger } from 'winston';
import { AppError } from '../errors/base-error';

interface ErrorResponse {
  ok: false;
  error: {
    code: string;
    message: string;
    details?: unknown;
    retryable: boolean;
    request_id?: string;
    trace_id?: string;
  };
  meta: {
    timestamp: string;
    duration_ms?: number;
  };
}

export function createErrorHandlerMiddleware(logger: Logger) {
  return (
    error: Error | AppError,
    req: Request,
    res: Response,
    _next: NextFunction
  ): void => {
    const startTime = res.locals.startTime || Date.now();
    const requestId = req.headers['x-request-id'] as string || 'unknown';
    const traceId = req.headers['x-trace-id'] as string || requestId;

    const appError = error instanceof AppError ? error : normalizeError(error);

    const response: ErrorResponse = {
      ok: false,
      error: {
        code: appError.errorCode,
        message: sanitizeErrorMessage(appError, req.app.get('env')),
        details: Object.keys(appError.context).length > 0 ? appError.context : undefined,
        retryable: appError.retryable,
        request_id: requestId,
        trace_id: traceId
      },
      meta: {
        timestamp: new Date().toISOString(),
        duration_ms: Date.now() - startTime
      }
    };

    const logLevel = selectLogLevel(appError.statusCode);
    logger.log(logLevel, 'Request error', {
      requestId,
      traceId,
      method: req.method,
      url: req.originalUrl,
      statusCode: appError.statusCode,
      errorCode: appError.errorCode,
      errorMessage: appError.message,
      errorStack: process.env.NODE_ENV === 'development' ? appError.stack : undefined,
      userIp: req.ip,
      userAgent: req.get('User-Agent')
    });

    res.status(appError.statusCode).json(response);
  };
}

function normalizeError(error: Error): AppError {
  if (error instanceof AppError) {
    return error;
  }

  if (error.type === 'entity.parse.failed') {
    return new AppError('Invalid JSON in request body', {
      statusCode: 400,
      errorCode: 'SYSTEM_INVALID_JSON',
      category: 'VALIDATION' as any,
      retryable: false
    });
  }

  if ('status' in error && typeof (error as any).status === 'number') {
    return new AppError(error.message, {
      statusCode: (error as any).status,
      errorCode: 'SYSTEM_HTTP_ERROR',
      category: 'INTERNAL' as any,
      retryable: (error as any).status >= 500
    });
  }

  return new AppError('Internal server error', {
    statusCode: 500,
    errorCode: 'SYSTEM_INTERNAL_ERROR',
    category: 'INTERNAL' as any,
    context: { originalMessage: error.message },
    retryable: false
  }, error);
}

function sanitizeErrorMessage(error: AppError, nodeEnv: string): string {
  if (nodeEnv === 'production' && error.statusCode >= 500) {
    return 'An unexpected error occurred. Please try again later.';
  }
  return error.message;
}

function selectLogLevel(statusCode: number): string {
  if (statusCode >= 500) return 'error';
  if (statusCode >= 400) return 'warn';
  return 'info';
}
```

## 5. 使用建议

### 5.1 何时使用这些模式

| 场景 | 推荐模式 | 参考章节 |
|------|----------|----------|
| 数据库 CRUD 操作 | Repository + Transaction + Validation | §1.1 |
| 外部 API 调用 | Retry with Backoff + Timeout + Error Classification | §2 |
| 配置读取 | Layered Loading + Validation + Hot Reload | §3 |
| HTTP 错误处理 | Unified Error Response + Sanitization | §4 |

### 5.2 关键原则总结

1. **Fail Fast**: 在入口处进行参数验证，尽早失败
2. **Never Swallow Errors**: 所有错误必须被记录或向上传播
3. **Graceful Degradation**: 外部故障时降级而非崩溃
4. **Context-Rich Logging**: 每条日志包含足够的上下文用于排查
5. **Resource Cleanup**: 使用 finally 或 try-with-resources 确保资源释放
6. **Security-Aware**: 不在响应中泄露内部细节
